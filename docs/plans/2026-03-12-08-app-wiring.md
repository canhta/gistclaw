# GistClaw Plan 8: App Wiring

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

// TestRunShutdownClean verifies that App.Run returns nil when the context is
// immediately cancelled (clean shutdown — no services have time to fail).
func TestRunShutdownClean(t *testing.T) {
	os.Clearenv()
	setMinimalEnv(t)

	cfg, err := config.Load()
	if err != nil {
		t.Fatalf("config.Load: %v", err)
	}

	a, err := app.NewApp(cfg)
	if err != nil {
		t.Fatalf("NewApp: %v", err)
	}

	ctx, cancel := context.WithCancel(context.Background())
	cancel() // cancel immediately

	done := make(chan error, 1)
	go func() { done <- a.Run(ctx) }()

	select {
	case err := <-done:
		if err != nil {
			t.Errorf("Run returned non-nil error on clean shutdown: %v", err)
		}
	case <-time.After(5 * time.Second):
		t.Fatal("Run did not return within 5s after context cancellation")
	}
}

// containsWarningEmoji returns true if s contains the Unicode warning sign ⚠️ (U+26A0).
func containsWarningEmoji(s string) bool {
	for _, r := range s {
		if r == '⚠' {
			return true
		}
	}
	return false
}

// internal/app/app.go
package app

import (
	"context"
	"errors"
	"fmt"
	"time"

	"golang.org/x/sync/errgroup"
	"github.com/rs/zerolog/log"

	"github.com/canhta/gistclaw/internal/agent"
	"github.com/canhta/gistclaw/internal/channel"
	"github.com/canhta/gistclaw/internal/channel/telegram"
	"github.com/canhta/gistclaw/internal/claudecode"
	"github.com/canhta/gistclaw/internal/config"
	"github.com/canhta/gistclaw/internal/gateway"
	"github.com/canhta/gistclaw/internal/hitl"
	"github.com/canhta/gistclaw/internal/infra"
	"github.com/canhta/gistclaw/internal/mcp"
	"github.com/canhta/gistclaw/internal/opencode"
	"github.com/canhta/gistclaw/internal/providers"
	"github.com/canhta/gistclaw/internal/scheduler"
	"github.com/canhta/gistclaw/internal/store"
	"github.com/canhta/gistclaw/internal/tools"
)

// App is the single composition root for GistClaw.
// All services are constructed by NewApp and wired here.
// Run executes the errgroup supervision policy.
type App struct {
	cfg        config.Config
	store      *store.Store
	channel    channel.Channel
	costGuard  *infra.CostGuard
	soul       *infra.SOULLoader
	heartbeat  *infra.Heartbeat
	llm        providers.LLMProvider // raw, undecorated
	trackedLLM providers.LLMProvider // decorated with trackingProvider
	opencode   opencode.Service
	claudecode claudecode.Service
	hitl       *hitl.Service
	scheduler  *scheduler.Service
	gateway    *gateway.Service
	mcp        *mcp.MCPManager
}

// NewApp constructs the full dependency graph.
// Returns an error if any required service fails to initialize.
//
// Construction order (each step may depend on the previous):
//  1. Open SQLite store
//  2. Run startup purges
//  3. Create infra: CostGuard, SOULLoader, Heartbeat
//  4. Create LLM provider + tracking decorator
//  5. Load MCP config + create MCPManager
//  6. Create search provider + web fetcher
//  7. Create channel (Telegram)
//  8. Create opencode.Service + claudecode.Service
//  9. Wire CostGuard.WithOperator (needs channel after step 7)
// 10. Create hitl.Service
// 11. Create scheduler.Service + wire JobTarget
// 12. Create gateway.Service
// 13. Create Heartbeat with agent checkers
func NewApp(cfg config.Config) (*App, error) {
	// ── Step 1: Open SQLite store ───────────────────────────────────────────
	s, err := store.New(cfg.SQLitePath)
	if err != nil {
		return nil, fmt.Errorf("app: store.New: %w", err)
	}

	// ── Step 2: Run startup purges ──────────────────────────────────────────
	// Remove expired cost history entries, stale HITL requests, etc.
	if err := s.PurgeExpired(cfg.Tuning.CostHistoryTTL, cfg.Tuning.SessionTTL); err != nil {
		// Non-fatal: log and continue. Stale rows waste space but don't break behaviour.
		log.Warn().Err(err).Msg("app: startup purge error (non-fatal)")
	}

	// ── Step 3: Infra ───────────────────────────────────────────────────────
	costGuard := infra.NewCostGuard(s, cfg.DailyLimitUSD)
	soul := infra.NewSOULLoader(cfg.SoulPath)
	// Heartbeat is created last (step 13) because it needs agent checkers.

	// ── Step 4: LLM provider + tracking decorator ───────────────────────────
	rawLLM, err := providers.New(cfg)
	if err != nil {
		return nil, fmt.Errorf("app: providers.New: %w", err)
	}
	trackedLLM := providers.NewTrackingProvider(rawLLM, costGuard)

	// ── Step 5: MCP config + MCPManager ────────────────────────────────────
	mcpCfg, err := mcp.LoadConfig(cfg.MCPConfigPath)
	if err != nil {
		// Non-fatal: MCP is optional. Log and use empty config.
		log.Warn().Err(err).Str("path", cfg.MCPConfigPath).Msg("app: mcp.LoadConfig error; MCP tools disabled")
		mcpCfg = &mcp.Config{}
	}
	mcpManager := mcp.NewManager(mcpCfg)

	// ── Step 6: Search provider + web fetcher ──────────────────────────────
	searchProvider := tools.NewSearchProvider(cfg)
	webFetcher := tools.NewWebFetcher(cfg.Tuning.WebFetchTimeout)

	// ── Step 7: Channel ─────────────────────────────────────────────────────
	ch, err := telegram.New(cfg.TelegramToken)
	if err != nil {
		return nil, fmt.Errorf("app: telegram.New: %w", err)
	}

	// ── Step 8: Agent services ──────────────────────────────────────────────
	soulText := soul.Load()
	ocSvc, err := opencode.New(cfg, s, costGuard, soulText)
	if err != nil {
		return nil, fmt.Errorf("app: opencode.New: %w", err)
	}
	ccSvc, err := claudecode.New(cfg, s, costGuard, soulText)
	if err != nil {
		return nil, fmt.Errorf("app: claudecode.New: %w", err)
	}

	// ── Step 9: Wire CostGuard operator notification (needs channel) ────────
	// CostGuard sends a budget-exceeded warning to the operator chat.
	costGuard.WithOperator(ch, cfg.OperatorChatID())

	// ── Step 10: HITL service ───────────────────────────────────────────────
	hitlSvc := hitl.New(s, ch, cfg.Tuning.HITLTimeout, cfg.Tuning.HITLReminderBefore)

	// ── Step 11: Scheduler + JobTarget ─────────────────────────────────────
	// appJobTarget is defined below. It is the adapter between the scheduler
	// (which works with abstract agent.Kind) and the concrete agent services.
	jobTarget := &appJobTarget{
		opencode:   ocSvc,
		claudecode: ccSvc,
		ch:         ch,
		operatorID: cfg.OperatorChatID(),
	}
	schedulerSvc := scheduler.NewService(s, jobTarget, cfg.Tuning)

	// ── Step 12: Gateway ────────────────────────────────────────────────────
	gatewaySvc := gateway.NewService(
		ch,
		hitlSvc,
		ocSvc,
		ccSvc,
		trackedLLM,
		searchProvider,
		webFetcher,
		mcpManager,
		schedulerSvc,
		s,
		cfg,
	)

	// ── Step 13: Heartbeat ──────────────────────────────────────────────────
	// Heartbeat checks agent liveness; it needs the constructed agent services.
	heartbeat := infra.NewHeartbeat(
		cfg.Tuning.HeartbeatTier1Every,
		cfg.Tuning.HeartbeatTier2Every,
		ch,
		cfg.OperatorChatID(),
		[]infra.AgentChecker{
			{Name: "opencode",   IsAlive: ocSvc.IsAlive},
			{Name: "claudecode", IsAlive: ccSvc.IsAlive},
		},
	)

	return &App{
		cfg:        cfg,
		store:      s,
		channel:    ch,
		costGuard:  costGuard,
		soul:       soul,
		heartbeat:  heartbeat,
		llm:        rawLLM,
		trackedLLM: trackedLLM,
		opencode:   ocSvc,
		claudecode: ccSvc,
		hitl:       hitlSvc,
		scheduler:  schedulerSvc,
		gateway:    gatewaySvc,
		mcp:        mcpManager,
	}, nil
}

// ── JobTarget adapter ────────────────────────────────────────────────────────

// appJobTarget implements scheduler.JobTarget.
// It uses the concrete agent services and channel to route scheduled jobs.
type appJobTarget struct {
	opencode   opencode.Service
	claudecode claudecode.Service
	ch         channel.Channel
	operatorID int64
}

// RunAgentTask routes a scheduled agent task to the appropriate agent service.
func (t *appJobTarget) RunAgentTask(ctx context.Context, kind agent.Kind, prompt string) error {
	switch kind {
	case agent.KindOpenCode:
		return t.opencode.SubmitTask(ctx, t.operatorID, prompt)
	case agent.KindClaudeCode:
		return t.claudecode.SubmitTask(ctx, t.operatorID, prompt)
	default:
		return fmt.Errorf("unknown agent kind: %v", kind)
	}
}

// SendChat sends a plain text message to the given chat ID.
func (t *appJobTarget) SendChat(ctx context.Context, chatID int64, text string) error {
	return t.ch.SendMessage(ctx, chatID, text)
}

// ── Run — supervision policy ─────────────────────────────────────────────────

// Run starts all services under an errgroup supervision policy.
//
// Supervision policy:
//   - gateway is CRITICAL: PermanentFailure → returns the error → errgroup cancels all
//     goroutines → process exits → systemd restarts the binary.
//   - opencode, claudecode, hitl, scheduler are NON-CRITICAL: PermanentFailure → send
//     operator notification → return nil (do NOT propagate the error to errgroup).
//
// Note: Go 1.22+ loop variable capture is fixed — no `svc := svc` needed.
func (a *App) Run(ctx context.Context) error {
	eg, ctx := errgroup.WithContext(ctx)

	// ── Critical: gateway ────────────────────────────────────────────────────
	// PermanentFailure from gateway propagates upward → cancels root context →
	// systemd/process supervisor restarts the binary.
	eg.Go(func() error {
		err := WithRestart("gateway", 0, 0, a.gateway.Run)(ctx)
		var pf PermanentFailure
		if errors.As(err, &pf) {
			log.Error().Str("service", pf.Name).Err(pf.Err).Msg("gateway permanently failed")
			return pf
		}
		if err != nil {
			log.Error().Str("service", "gateway").Err(err).Msg("BUG: unexpected error from WithRestart")
			return err
		}
		return nil
	})

	// ── Non-critical services ────────────────────────────────────────────────
	// PermanentFailure does NOT propagate to errgroup — operator is notified
	// and the binary continues running with degraded capabilities.
	type nonCritical struct {
		name        string
		maxAttempts int
		window      time.Duration
		fn          func(context.Context) error
	}
	services := []nonCritical{
		{"opencode",   5, 30 * time.Second, a.opencode.Run},
		{"claudecode", 5, 30 * time.Second, a.claudecode.Run},
		{"hitl",       10, 10 * time.Second, a.hitl.Run},
		{"scheduler",  5, 30 * time.Second, a.scheduler.Run},
	}
	for _, svc := range services {
		eg.Go(func() error {
			err := WithRestart(svc.name, svc.maxAttempts, svc.window, svc.fn)(ctx)
			var pf PermanentFailure
			if errors.As(err, &pf) {
				log.Error().Str("service", pf.Name).Err(pf.Err).Msg("permanently stopped")
				if pf.Name == "hitl" {
					a.hitl.DrainPending()
				}
				// Notify operator — best-effort, do not block or error on failure.
				_ = a.channel.SendMessage(ctx, a.cfg.OperatorChatID(), FormatDegradedMsg(pf.Name))
				return nil // do NOT cancel root context
			}
			if err != nil {
				log.Error().Str("service", svc.name).Err(err).Msg("BUG: unexpected error from WithRestart")
			}
			return nil
		})
	}

	// ── Heartbeat ────────────────────────────────────────────────────────────
	eg.Go(func() error {
		a.heartbeat.Run(ctx)
		return nil
	})

	return eg.Wait()
}

// FormatDegradedMsg returns the operator notification message for a degraded service.
// Exported for testing.
func FormatDegradedMsg(name string) string {
	switch name {
	case "opencode":
		return "⚠️ OpenCode is unavailable. Use /stop then try /oc again."
	case "claudecode":
		return "⚠️ Claude Code is unavailable. Use /stop then try /cc again."
	case "hitl":
		return "⚠️ HITL unavailable. Approval requests will be auto-rejected."
	case "scheduler":
		return "⚠️ Scheduler unavailable. Scheduled jobs are paused."
	default:
		return fmt.Sprintf("⚠️ Service %q is unavailable.", name)
	}
}
