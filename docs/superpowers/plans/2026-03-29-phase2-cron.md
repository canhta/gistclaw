# Cron Jobs Section Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Implement the `/cron` section showing scheduled jobs (Jobs tab), recent run occurrences (Runs tab), and a job creation/edit form (Editor tab).

**Architecture:** The page loads from `GET /api/automate`. Jobs tab shows the schedule list with enable/disable/run-now actions via `POST /api/automate/{id}/enable`, `POST /api/automate/{id}/disable`, `POST /api/automate/{id}/run`. Runs tab shows open + recent occurrences. Editor tab is a form for creating new schedules via `POST /api/automate`. State for action results is local (`$state`). Page data is typed via `+page.ts`.

**Tech Stack:** SvelteKit, Svelte 5 runes, `$lib/automate/load.ts` (existing), `$lib/types/api.ts` (existing), `$lib/http/client.ts` for POSTs, Vitest for tests, Warm Brutalism CSS.

import { loadAutomate } from '$lib/automate/load';

export const load: PageLoad = async ({ fetch }) => {
  try {
    const data = await loadAutomate(fetch);
    return {
      cron: {
        schedules: data.schedules ?? [],
        openOccurrences: data.open_occurrences ?? [],
        recentOccurrences: data.recent_occurrences ?? [],
        summary: data.summary,
      }
    };
  } catch {
    return {
      cron: {
        schedules: [],
        openOccurrences: [],
        recentOccurrences: [],
        summary: null,

};

<!-- src/routes/cron/+page.svelte -->
<script lang="ts">
  import SectionTabs from '$lib/components/shell/SectionTabs.svelte';
  import ScheduleRow from '$lib/components/cron/ScheduleRow.svelte';
  import OccurrenceRow from '$lib/components/cron/OccurrenceRow.svelte';
  import { requestJSON } from '$lib/http/client';
  import type { PageData } from './$types';

  let { data }: { data: PageData } = $props();

  const tabs = [
    { id: 'jobs', label: 'Jobs' },
    { id: 'runs', label: 'Runs' },
    { id: 'editor', label: 'Editor' },
  ];
  let activeTab = $state('jobs');

  const schedules = $derived(data.cron?.schedules ?? []);
  const openOccurrences = $derived(data.cron?.openOccurrences ?? []);
  const recentOccurrences = $derived(data.cron?.recentOccurrences ?? []);

  let actionError = $state('');
  let actionMessage = $state('');

  // Editor form state
  let editorName = $state('');
  let editorObjective = $state('');
  let editorCadence = $state('');
  let editorSubmitting = $state(false);
  let editorError = $state('');

  async function handleEnable(id: string): Promise<void> {
    actionError = '';
    actionMessage = '';
    try {
      await requestJSON(fetch, `/api/automate/${id}/enable`, { method: 'POST' });
      actionMessage = 'Schedule enabled.';
    } catch {
      actionError = 'Failed to enable schedule.';
    }
  }

  async function handleDisable(id: string): Promise<void> {
    actionError = '';
    actionMessage = '';
    try {
      await requestJSON(fetch, `/api/automate/${id}/disable`, { method: 'POST' });
      actionMessage = 'Schedule disabled.';
    } catch {
      actionError = 'Failed to disable schedule.';
    }
  }

  async function handleRunNow(id: string): Promise<void> {
    actionError = '';
    actionMessage = '';
    try {
      await requestJSON(fetch, `/api/automate/${id}/run`, { method: 'POST' });
      actionMessage = 'Job dispatched.';
    } catch {
      actionError = 'Failed to run schedule.';
    }
  }

  async function handleCreate(e: SubmitEvent): Promise<void> {
    e.preventDefault();
    editorError = '';
    editorSubmitting = true;
    try {
      await requestJSON(fetch, '/api/automate', {
        method: 'POST',
        headers: { 'content-type': 'application/json' },
        body: JSON.stringify({
          name: editorName,
          objective: editorObjective,
          cadence: editorCadence,
        }),
      });
      editorName = '';
      editorObjective = '';
      editorCadence = '';
      actionMessage = 'Schedule created.';
      activeTab = 'jobs';
    } catch {
      editorError = 'Failed to create schedule.';
    } finally {
      editorSubmitting = false;
    }
  }
</script>

<svelte:head>
  <title>Cron Jobs | GistClaw</title>
</svelte:head>

<div class="flex h-full flex-col overflow-hidden">
  <div class="shrink-0 border-b border-b-[1.5px] border-[var(--gc-border-strong)] bg-[var(--gc-surface)]">
    <div class="px-6 pt-4 pb-0">
      <h1 class="gc-panel-title text-[var(--gc-ink)]">Cron Jobs</h1>
    </div>
    <SectionTabs {tabs} bind:activeTab />
  </div>

  <div class="flex min-h-0 flex-1 flex-col overflow-y-auto">
    {#if actionMessage}
      <div class="border-b border-b-[1.5px] border-[var(--gc-success)] bg-[var(--gc-success-dim)] px-5 py-3">
        <p class="gc-copy text-[var(--gc-success)]">{actionMessage}</p>
      </div>
    {/if}
    {#if actionError}
      <div class="border-b border-b-[1.5px] border-[var(--gc-error)] bg-[var(--gc-error-dim)] px-5 py-3">
        <p class="gc-copy text-[var(--gc-error)]">{actionError}</p>
      </div>
    {/if}

    {#if activeTab === 'jobs'}
      {#if schedules.length === 0}
        <div class="flex flex-1 items-center justify-center p-10">
          <div class="text-center">
            <p class="gc-stamp text-[var(--gc-ink-3)]">CRON</p>
            <p class="gc-panel-title mt-3 text-[var(--gc-ink)]">No scheduled jobs</p>
            <p class="gc-copy mt-3 max-w-xs text-[var(--gc-ink-2)]">
              Create a schedule to run tasks automatically. Use the Editor tab to get started.
            </p>
          </div>
        </div>
      {:else}
        {#each schedules as schedule (schedule.id)}
          <ScheduleRow
            {schedule}
            onEnable={handleEnable}
            onDisable={handleDisable}
            onRunNow={handleRunNow}
          />
        {/each}
      {/if}
    {:else if activeTab === 'runs'}
      {#if openOccurrences.length === 0 && recentOccurrences.length === 0}
        <div class="flex flex-1 items-center justify-center p-10">
          <div class="text-center">
            <p class="gc-stamp text-[var(--gc-ink-3)]">RUNS</p>
            <p class="gc-panel-title mt-3 text-[var(--gc-ink)]">No runs yet</p>
            <p class="gc-copy mt-3 max-w-xs text-[var(--gc-ink-2)]">
              Run history will appear here after schedules execute.
            </p>
          </div>
        </div>
      {:else}
        {#if openOccurrences.length > 0}
          <div class="border-b border-b-[1.5px] border-[var(--gc-border)] px-5 py-3">
            <p class="gc-stamp text-[var(--gc-ink-3)]">ACTIVE</p>
          </div>
          {#each openOccurrences as occ (occ.id)}
            <OccurrenceRow occurrence={occ} />
          {/each}
        {/if}
        {#if recentOccurrences.length > 0}
          <div class="border-b border-b-[1.5px] border-[var(--gc-border)] px-5 py-3">
            <p class="gc-stamp text-[var(--gc-ink-3)]">RECENT</p>
          </div>
          {#each recentOccurrences as occ (occ.id)}
            <OccurrenceRow occurrence={occ} />
          {/each}
        {/if}
      {/if}
    {:else}
      <!-- Editor tab -->
      <div class="px-6 py-6 max-w-lg">
        <p class="gc-stamp text-[var(--gc-ink-3)]">NEW JOB</p>
        <h2 class="gc-panel-title mt-3 text-[var(--gc-ink)]">Create a schedule</h2>

        {#if editorError}
          <div class="mt-4 border border-[1.5px] border-[var(--gc-error)] bg-[var(--gc-error-dim)] px-4 py-3">
            <p class="gc-copy text-[var(--gc-error)]">{editorError}</p>
          </div>
        {/if}

        <form onsubmit={handleCreate} class="mt-6 flex flex-col gap-4">
          <div>
            <label class="gc-stamp text-[var(--gc-ink-3)]" for="cron-name">NAME</label>
            <input
              id="cron-name"
              type="text"
              bind:value={editorName}
              required
              placeholder="Daily digest"
              class="gc-input mt-1 w-full px-3 py-2"
            />

            <label class="gc-stamp text-[var(--gc-ink-3)]" for="cron-objective">OBJECTIVE</label>

  </div>
</div>
